PROCEDURE ADDACCOUNTCOMMENT
 FMPERFORM CREATE COMMENT LOC AFTERLAST (1,0,ERRORTEXT)
 DO
  SET EFFECTIVEDATE TO SYSTEMDATE
  SET COMMENT TO COMMENTTEXT
 END
END

PROCEDURE EXPIREACCOUNTCOMMENT
 FOR EACH COMMENT WITH(COMMENT:COMMENT=COMMENTTEXT)
 DO
  COMMENTLOC=COMMENT:LOCATOR
 END
 FMPERFORM REVISE COMMENT LOC COMMENTLOC (1,1,ERRORTEXT)
 DO
  SET EXPIRATIONDATE TO SYSTEMDATE
 END
END
